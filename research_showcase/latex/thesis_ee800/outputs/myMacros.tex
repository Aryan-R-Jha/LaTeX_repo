\newcommand{\gen}[1]{% #1 is the function symbol, e.g., f or L
    \expandafter\newcommand\csname #1k\endcsname{#1(x_k, u_k, k)}
    \expandafter\newcommand\csname #1bark\endcsname{#1(\bar{x}_k, \bar{u}_k, k)}
    \expandafter\newcommand\csname #1kp\endcsname{#1(x_{k+1}, u_{k+1}, k+1)}
    \expandafter\newcommand\csname #1barkp\endcsname{#1(\bar{x}_{k+1}, \bar{u}_{k+1}, k+1)}
    \expandafter\newcommand\csname #1j\endcsname{#1(x_j, u_j, j)}

    \expandafter\newcommand\csname #1kdelx\endcsname{#1_x \delta x_k}
    \expandafter\newcommand\csname #1kdelxx\endcsname{\frac{1}{2} #1_{xx} (\delta x_k)^2}
    \expandafter\newcommand\csname #1kdelu\endcsname{#1_u \delta u_k}
    \expandafter\newcommand\csname #1kdeluu\endcsname{\frac{1}{2} #1_{uu} (\delta u_k)^2}
    \expandafter\newcommand\csname #1kdelxu\endcsname{\frac{1}{2} #1_{xu} (\delta x_k) (\delta u_k)}

    \expandafter\newcommand\csname #1kpdelx\endcsname{#1_x \delta x_{k+1}}
    \expandafter\newcommand\csname #1kpdelxx\endcsname{\frac{1}{2} #1_{xx} (\delta x_{k+1})^2}
    \expandafter\newcommand\csname #1kpdelu\endcsname{#1_u \delta u_{k+1}}
    \expandafter\newcommand\csname #1kpdeluu\endcsname{\frac{1}{2} #1_{uu} (\delta u_{k+1})^2}
    \expandafter\newcommand\csname #1kpdelxu\endcsname{\frac{1}{2} #1_{xu} (\delta x_{k+1}) (\delta u_{k+1})}
}

\gen{f} % Generate commands for f
\gen{L} % Generate commands for L
