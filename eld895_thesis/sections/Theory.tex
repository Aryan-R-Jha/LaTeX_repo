\begin{frame}{AR1 and Variance}
	\begin{equation}
		\label{eq:autocorrDef}
		d[k] = a_1 d[k-1] + e[k]
	\end{equation} 
	Autoregressive Model of Order 1 on detrended voltages.
	
	\begin{equation}
		\label{eq:varDef}
		\sigma^2 = \frac{1}{n_k} \sum_{k=1}^{n_k} d[k]^2
	\end{equation} 
	Sample variance on detrended voltages.
\end{frame}
