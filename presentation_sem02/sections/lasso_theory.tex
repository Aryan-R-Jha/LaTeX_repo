\begin{frame}[fragile]{State Space Representation}
	Express Power System Dynamics in State Space:
	\begin{equation}
		\label{eq:ssr}
		\begin{multiline}
			\dot{\textbf{x}}(t) = 
			\textbf{A}\textbf{x}(t)
			+ \textbf{B}\textbf{u}(t)\\
			\textbf{y}(t) = 
			\textbf{C}\textbf{x}(t)  
			\forall t\geq0
		\end{multiline}
	\end{equation}
\end{frame}
